var tomb = [2, 3, 4, 1, 5, 6, 7, 10, 8, 9];
var meret = tomb.length;

var min = tomb[0];


for (i = 0; i < meret; i++)
  if (tomb[i] < min)
    min = tomb[i];
alert('minimum:');

var max = 0;

for (i = 0; i < meret; i++)
  if (tomb[i] > max)
    max = tomb[i];
alert('maximum:');
alert(max);
